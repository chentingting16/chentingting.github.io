选择排序

def SelectionSort(nums: List[int]):
n = len(nums)
#每次循环从[i,n]中选出最小的，交换至i位置
for i in range(n):
    min_index = i
    for j in range(i,n):
        if nums[j]<nums[min_index]:
            min_index = j
    nums[i],nums[min_index] = nums[min_index],nums[i]

插入排序

def InsertSort(nums:List[int]):
n = len(nums)
#每次循环从[0,i-1]中选择插入的位置为x，然后将x之后的已排序数字向后移一个
for i in range(1,n):
    insertNum = nums[i]
    j = i-1
    while j>=0 and insertNum<nums[j]:
        nums[j+1] = nums[j]
        j -= 1
    nums[j+1] = insertNum
    print(nums)
